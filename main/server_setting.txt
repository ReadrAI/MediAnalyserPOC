[Unit]
Description=Gunicorn instance to serve newshorizonapp
After=network.target

[Service]
User=jean
Group=www-data
WorkingDirectory=$(REPOPATH)/flask_app
Environment="PATH=/usr/bin/python"
EnvironmentFile=$(REPOPATH)/flask_app/.env
ExecStart=/usr/local/bin/gunicorn --workers 3 --bind unix:/var/run/newshorizon.sock --pythonpath $(REPOPATH) wsgi:app


[Install]
WantedBy=multi-user.target
